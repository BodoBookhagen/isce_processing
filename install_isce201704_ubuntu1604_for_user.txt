### setup ISCE on an installed system
cd ~
wget https://repo.continuum.io/miniconda/Miniconda3-latest-Linux-x86_64.sh
bash Miniconda3-latest-Linux-x86_64.sh
# hit return at each option/choose default options

### Install environment for isce
#copy isce_201704.yml to your home directory (or elsewhere)
#isce_201704.yml is a modified version from David Bekaert's github site https://github.com/dbekaert in directory isce_notes/Ubuntu/isce_201609.yml

cd ~
miniconda3/bin/conda env create -f isce_201704.yml

# Setup .isce:
cd ~
mkdir .isce
# copy .isceenv_20170403 to ~/.isce

### Add aliases to ~/.bashrc
alias start_isce="source activate isce_201704; source ~/.isce/.isceenv_20170403"

### Start ISCE environment with:
start_isce
